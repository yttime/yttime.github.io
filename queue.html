QUEUE 



#include <stdio.h>

int queue[10],rear,front;

int isEmpty(){
	return rear==front;
}
void enqueue(int item){
	   queue[rear++]=item;
}

int dequeue(){
	   if(rear>0)
	       return queue[front++];
	   else
	       return 0;
}

int main(){
	printf("%d\n",dequeue());
	enqueue(10);
	enqueue(12);
	enqueue(13);
	printf("%d\n",dequeue());
	printf("%d\n",dequeue());
	printf("%d\n",dequeue());
		
	return 0;
}
