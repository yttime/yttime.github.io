Stack data structure

Stack is last in last out. Five operations. peek(), push(), pop(), size(), isempty(). 


#include <iostream>
using namespace std;

int stk[100], top=-1,sz=99;

bool isempty(){
	return top== -1;
}

int peek(){
	return stk[top];
}
void push(int data){
	if (top == sz)
	    cout<<"stack is overflow\n";
	else
	   stk[++top]=data;
}

void pop(){
	if (top<0)
	   cout<<"stack is underflow\n";
	else
	   --top;
}

int size(){
	return top+1;
}
	
	 



}
	
	 


int main(){
	
	push(100);
	cout<<"top is: "<< peek()<<endl;
	push(2);
	push(3);
	cout<<size();
	pop();
	cout<<"top is: "<< peek()<<endl;
		
	pop();
	cout<<"top is: "<< peek()<<endl;
	
	pop();
	pop();
	isempty()?cout<<"empty":cout<<"not empty";
	
	return 0;
}
